 vector<vector<int> > subsetsWithDup(vector<int> &S) {
        sort(S.begin(), S.end());
        vector<vector<int>> result(1);
        int oldval=S[0];
        int oldj=0;
        for(int i=0; i<S.size(); i++){
            int temp=oldj;
            if(S[i]!=oldval){
                oldval=S[i]; temp=0;
            }
            int j=result.size();
            oldj=j;
            while(j-->temp){
                //note temp here help avoid creating duplicate subsets
                result.push_back(result[j]);
                result.back().push_back(S[i]);
            }
        }
        return result;
    }
