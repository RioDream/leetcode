##Largest Rectangle in Histogram 


8-12

class Solution {
public:
    int largestRectangleArea(vector<int> &height) {
        /*
        右边界
        左边界
        */
    	const int n = height.size();
    	stack<int> s;
    	stack<int> len;
    
    	height.push_back(0);
    	s.push(height[0]);
    	len.push(1);

    	int maxArea = 0;
    	for (int i = 1; i <= n; i++){
    	    //找右边界
    		if (height[i] < height[i - 1]){
    			int length = 0;
    			//找左边界
    			while (!s.empty() && s.top()>height[i]){
    				int h = s.top();
    				int l = len.top();
    				length += l;
    				s.pop();
    				len.pop();
    
    				maxArea = max(maxArea, h*length);
    			}
    			
    			len.push(length + 1);
    			s.push(height[i]);
    		}
    		else{
    			s.push(height[i]);
    			len.push(1);
    		}
    	}
    	return maxArea;
    }
    
};
